52static inline int vod_MediaControl( vod_t *p_vod, vod_media_t *p_media,
53 const char *psz_id, int i_query, ... )
54{
55 va_list args;
56 int i_result;
57
58 if( !p_vod->pf_media_control ) return VLC_EGENERIC;
59
60 va_start( args, i_query );
61 i_result = p_vod->pf_media_control( p_vod->p_data, p_media, psz_id,
62 i_query, args );
63 va_end( args );
