242static inline const char *config_CategoryNameGet( int i_value )
243{
244 int i = 0;
245 while( categories_array[i].psz_name != NULL )
246 {
247 if( categories_array[i].i_id == i_value )
248 {
249 return vlc_gettext(categories_array[i].psz_name);
250 }
251 i++;
252 }
253 return NULL;
254}
