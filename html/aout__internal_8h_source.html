167static inline void aout_SetWavePhysicalChannels(audio_sample_format_t *fmt)
168{
169 static const uint32_t wave_channels[] = {
170 AOUT_CHAN_LEFT, AOUT_CHAN_RIGHT, AOUT_CHAN_CENTER,
171 AOUT_CHAN_LFE, AOUT_CHAN_REARLEFT, AOUT_CHAN_REARRIGHT,
172 AOUT_CHAN_MIDDLELEFT, AOUT_CHAN_MIDDLERIGHT, AOUT_CHAN_REARCENTER };
173
174 fmt->i_physical_channels = 0;
175 for (int i = 0; i < fmt->i_channels && i < AOUT_CHAN_MAX; ++i)
176 fmt->i_physical_channels |= wave_channels[i];
177 aout_FormatPrepare(fmt);
178}
